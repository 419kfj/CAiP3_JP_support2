2.1.1 TとCの平均の差

mean(.Test) - mean(.Ctrl)
## [1] 1.809091
set.seed(123)
.Data_all <- c(.Test,.Ctrl)
.Data_idx <- 1:25
sample(.Data_idx,size = 14,replace = FALSE) -> set_T
.Data_idx[!(.Data_idx %in% set_T)] -> set_C
#set_T %>% length()
#set_C %>% length()
rn <- 9999
.Data_all <- c(.Test,.Ctrl)
.Data_idx <- 1:25
diff_mean <- double()
for (i in 1:rn){
#    print (i)
    sample(.Data_idx,size = 14,replace = FALSE) -> set_T
    .Data_idx[!(.Data_idx %in% set_T)] -> set_C
#    print(set_T,set_C)
    diff_mean.new <- mean(.Data_all[set_T]) - mean(.Data_all[set_C])
    diff_mean <- append(diff_mean,diff_mean.new)
}
diff_mean <- append(diff_mean,mean(.Test) - mean(.Ctrl))
hist(diff_mean,breaks = 20,main="Exhibit 30.2 n = 9999+1")
abline(v = mean(.Test) - mean(.Ctrl),col="blue")
