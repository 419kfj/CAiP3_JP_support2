ll <- NULL
for(i in 1:24){
  ll <- append(ll,.tmp[i,])
}
cg_levels <- ll

#.d.conv %>% mutate(CG=str_c(C,G),CGA=str_c(C,G,A)) -> .d.conv2
.d.conv %>%  count(C,G,Q3) %>% mutate(CG=factor(str_c(C,G),levels = cg_levels)) %>%
                                         filter(G!=9) %>% select(CG,Q3,n) %>% 
                                         spread(key=Q3,value = n) %>% # same as by xtabs
                                         mutate(Total=(W+w+H+`?`)) %>% 
                                         mutate(Wp=round(100*W/Total,1),wp=round(100*w/Total,1),
                                                Hp=round(100*H/Total,1),
                                                `?p`=round(100*`?`/Total,1)) %>%
                                         select(CG,Wp,wp,Hp,`?p`,Total)-> EX16.6.0
